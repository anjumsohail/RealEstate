SET @radius_km = 5000;      -- radius in meters (e.g. 5000m = 5km)

Query 1: Find cities that CONTAIN the point using MBR
sql
SELECT 
    city_id,
    city_name,
    ST_AsText(boundary) as boundary_wkt
FROM cities 
WHERE MBRContains(
    boundary, 
    ST_GeomFromText('POINT(longitude latitude)', 4326)
);
Query 2: Radius search using MBRContains with buffer
sql
SELECT 
    city_id,
    city_name,
    ST_AsText(boundary) as boundary_wkt,
    ST_Distance(
        ST_Centroid(boundary),
        ST_GeomFromText('POINT(longitude latitude)', 4326)
    ) * 111.325 as distance_km
FROM cities 
WHERE MBRContains(
    ST_Buffer(
        ST_GeomFromText('POINT(longitude latitude)', 4326),
        radius_km / 111.325
    ),
    boundary
);
Query 3: Alternative using MBRIntersects for radius search
sql
SELECT 
    city_id,
    city_name,
    ST_AsText(boundary) as boundary_wkt,
    ST_Distance(
        ST_Centroid(boundary),
        ST_GeomFromText('POINT(longitude latitude)', 4326)
    ) * 111.325 as distance_km
FROM cities 
WHERE MBRIntersects(
    boundary,
    ST_Buffer(
        ST_GeomFromText('POINT(longitude latitude)', 4326),
        radius_km / 111.325
    )
);
Query 4: Parameterized version with exact distance calculation
sql
SET @latitude = 40.7128;
SET @longitude = -74.0060;
SET @radius_km = 10;
SET @radius_degrees = @radius_km / 111.325;

SELECT 
    city_id,
    city_name,
    ST_AsText(boundary) as boundary_wkt,
    (ST_Distance(
        ST_Centroid(boundary),
        ST_GeomFromText(CONCAT('POINT(', @longitude, ' ', @latitude, ')'), 4326)
    ) * 111.325) as distance_km
FROM cities 
WHERE MBRContains(
    ST_Buffer(
        ST_GeomFromText(CONCAT('POINT(', @longitude, ' ', @latitude, ')'), 4326),
        @radius_degrees
    ),
    boundary
)
HAVING distance_km <= @radius_km
ORDER BY distance_km;
Query 5: More efficient two-step approach (MBR filter + precise check)
sql
SET @point = ST_GeomFromText('POINT(longitude latitude)', 4326);
SET @buffer = ST_Buffer(@point, radius_km / 111.325);

-- First filter by MBR (uses spatial index)
SELECT 
    city_id,
    city_name,
    ST_AsText(boundary) as boundary_wkt,
    ST_Distance(ST_Centroid(boundary), @point) * 111.325 as distance_km
FROM cities 
WHERE MBRIntersects(boundary, @buffer)
-- Then precise distance check
HAVING distance_km <= radius_km
ORDER BY distance_km;
Query 6: Simple point-in-polygon check with MBRWithin
sql
SELECT 
    city_id,
    city_name
FROM cities 
WHERE MBRWithin(
    ST_GeomFromText('POINT(longitude latitude)', 4326),
    boundary
)
AND ST_Contains(boundary, ST_GeomFromText('POINT(longitude latitude)', 4326));

DELIMITER //

CREATE FUNCTION to_GeoJSON(wkt_geometry TEXT) RETURNS TEXT
DETERMINISTIC
READS SQL DATA
BEGIN
    DECLARE geojson TEXT;
    DECLARE clean_coords TEXT;
    DECLARE formatted TEXT DEFAULT '';
    DECLARE pos INT;
    DECLARE pair TEXT;
    
    -- Extract coordinates from WKT
    SET clean_coords = REPLACE(REPLACE(wkt_geometry, 'POLYGON((', ''), '))', '');
    
    -- Process each coordinate pair
    WHILE LENGTH(clean_coords) > 0 DO
        SET pos = LOCATE(',', clean_coords);
        IF pos = 0 THEN
            SET pair = clean_coords;
            SET clean_coords = '';
        ELSE
            SET pair = SUBSTRING(clean_coords, 1, pos - 1);
            SET clean_coords = SUBSTRING(clean_coords, pos + 1);
        END IF;
        
        -- Format the coordinate pair
        IF LENGTH(formatted) > 0 THEN
            SET formatted = CONCAT(formatted, ',');
        END IF;
        
        SET formatted = CONCAT(formatted, '[', REPLACE(TRIM(pair), ' ', ','), ']');
    END WHILE;
    
    -- Build the complete GeoJSON
    SET geojson = CONCAT(
        '{"type":"Feature","geometry":',
        '{"type":"Polygon","coordinates":[',
        '[', formatted, ']',
        ']},"properties":{}',
        '}'
    );
    
    RETURN geojson;
END //

DELIMITER ;
